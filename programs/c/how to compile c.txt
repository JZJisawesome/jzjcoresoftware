//Do this for crt0.s
riscv64-unknown-elf-gcc -ffreestanding -nostdlib -march=rv32i -mabi=ilp32 -nostartfiles -nodefaultlibs -Os -c -mstrict-align crt0.s -o crt0.o

//Link everything together using jzjcorea.ld
